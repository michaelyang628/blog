确保APT软件仓库列表和已安装的软件包是最新的。

检查可用的软件包更新 sudo apt update

安装可用的软件包更新 sudo apt upgrade

query ip sudo apt install net-tools ifconfig

nmap 网络嗅探器 sudo apt install nmap 扫描局域网主机发现 sudo nmap -sP 192.168.45.0/24

打印所有服务状态 sudo service --status-all

查看整个系统磁盘使用情况 df -h 列出所有可用block设备的信息 lsblk 打印已安装包列表 dpkg -l

列出所有硬件 sudo lshw -short

配置静态IP地址
#

sudo nano /etc/netplan/50-cloud-init.yaml

ethernets:
    ens33:
      dhcp4: false
      addresses:
        - 192.168.1.100/24
      routes:
        - to: default
          via: 192.168.1.1
      nameservers:
        addresses:
          - 8.8.8.8
          - 114.114.114.114

sudo netplan apply

生成SSH KEY
#

ls -al ~/.ssh
ssh-keygen -t ed25519 -C "email"
cat ~/.ssh/id_ed25519.pub

磁盘 RAID
#

sudo mdadm --zero-superblock /dev/sdb
sudo mdadm --zero-superblock /dev/sdc

# 创建 RAID 0 阵列 /dev/md0
sudo mdadm --create /dev/md0 --level=0 --raid-devices=2 /dev/sdb /dev/sdc

ZFS（Zettabyte File System）
#

由Sun Microsystems开发的第一个128位文件系统，最初用于Solaris 10操作系统。 ZFS的设计目标是提供一个高扩展性、强大且易于管理的文件系统。

  • 存储池 —— 与传统文件系统不同，ZFS将所有设备集中到一个存储池中进行管理，这使得文件系统不再局限于单独的物理设备。
  • 写时拷贝事务模型 —— 含有活动数据的块从不被覆盖，而是分配一个新块，并将修改后的数据写在新块上。
  • 快照与克隆 —— 快照是文件系统在某一时刻的只读副本，而克隆是可写的快照。
  • 动态条带化 —— 当额外的设备被加入到存储池中时，条带宽度会自动扩展以包含这些设备。
  • 数据完整性与自我修复 —— 它对所有数据进行256位的校验，并在发现错误时自动修复。
  • 高效的数据压缩 —— 可以提供极快的压缩和解压缩速度。

用例
#

# Ubuntu 24.04 已内置 OpenZFS（2.2+），只需安装用户态工具
# 安装 ZFS 工具
sudo apt install -y zfsutils-linux
# 创建 ZFS 存储池（zpool）
sudo zpool create -f -o ashift=12 games sdb sdc sdd sde
# 查看池状态
sudo zpool status games
# 创建 ZVOL
sudo zfs create -V 10T -o volblocksize=4K games/steam
# 查看挂载点
zfs list

创建游戏仓库
#

# 进入 iSCSI 配置的命令行界面
sudo targetcli
# 将磁盘阵列 /dev/md0 定义为一个存储设备：
/backstores/block create steam_disk /dev/zvol/games/steam
# 创建 iSCSI Target (IQN)
/iscsi create
# 创建 LUN (逻辑单元号)
/tpg1/luns create /backstores/block/steam_disk

# 配置 ACL (访问控制列表)
tpg1/acls create

# 启用“自动生成 ACL”
set attribute generate_node_acls=1

# 禁用身份验证
set attribute authentication=0

saveconfig

exit

Nginx
#

概述
#

Nginx是一个开源web服务器，通常用作反向代理或HTTP缓存。 Linux免费提供。

安装Nginx
#

sudo apt install -y nginx

管理Nginx
#

  • 检查Nginx状态： systemctl status nginx
  • 停止Web服务器： sudo systemctl stop nginx
  • 启动Web服务器： sudo systemctl start nginx
  • 再次启动该服务： sudo systemctl restart nginx
  • 热加载： sudo systemctl reload nginx
  • 禁止Nginx： sudo systemctl disable nginx
  • 激活Nginx： sudo systemctl enable nginx
  • 检查Nginx配置文件： sudo nginx -t

为 Nextcloud 创建数据库和用户
#

CREATE DATABASE nextcloud;
CREATE USER 'nextcloud'@'localhost' IDENTIFIED BY 'your_password';
GRANT ALL PRIVILEGES ON nextcloud.* TO 'nextcloud'@'localhost';
FLUSH PRIVILEGES;
EXIT;

添加默认用户
#

CREATE USER 'yang'@'localhost' IDENTIFIED BY 'password';
grant all on *.* to yang@'localhost';
SHOW GRANTS FOR 'yang'@'localhost';
revoke all on *.* from yang@'localhost';
select user();
mysql -u yang -p

qBittorrent
#

安装
#

sudo add-apt-repository ppa:qbittorrent-team/qbittorrent-stable
sudo apt update
sudo apt install qbittorrent-nox

创建专用用户与目录
#

sudo adduser --system --group qbittorrent
sudo mkdir -p /mnt/nas/downloads/{movies,tv,software}
sudo chown -R qbittorrent:qbittorrent /mnt/nas/downloads

配置 systemd 服务
#

sudo nano /etc/systemd/system/qbittorrent.service

[Unit]
Description=qBittorrent BitTorrent client
After=network.target

[Service]
Type=exec
User=qbittorrent
Group=qbittorrent
ExecStart=/usr/bin/qbittorrent-nox --webui-port=8081
Restart=always
RestartSec=5

[Install]
WantedBy=multi-user.target

启用服务
#

sudo systemctl daemon-reload
sudo systemctl enable --now qbittorrent

Docusaurus
#

默认页面位于**/var/www/html/**位置。 您可以将静态页面放在此处，或者使用虚拟主机并将其放在其他位置。

虚拟主机是一种在同一服务器上托管多个域名的方法。 Docusaurus是一款现代化的静态网页生成器。 使用命令行工具可以帮助你快速简单地安装Docusaurus并搭建网站框架。

cd /var/www
npx create-docusaurus@latest tutorials classic

我们需要将网站生成为静态内容，并上传到网络服务器，才能被其他人访问。 要构建站点，请使用以下命令： cd tutorials npm run build

设置虚拟主机
#

要设置虚拟主机，我们需要在**/etc/nginx/sites-enabled/**目录中创建文件。

对于本教程，我们将在81端口上提供我们的站点，而不是标准的80端口。 如果您愿意，可以更改它。

cd /etc/nginx/sites-enabled
sudo "${EDITOR:-vi}" tutorials

server {
       listen 81;
       listen [::]:81;

       server_name docs.minebox.xyz;

       root /var/www/tutorials/build;
       index index.html;

       location / {
               try_files $uri $uri/ =404;
       }
}

root是放置our.html文件的目录。 index用于指定访问站点根目录时可用的文件。 server_name可以是您想要的任何内容，因为您现在还没有将它指向任何真实的域。 通过在sites-enabled目录新建一个链接 sudo ln -s /etc/nginx/sites-available/docs.minebox.xyz.conf /etc/nginx/sites-enabled/

激活虚拟主机并测试结果
#

要使我们的站点工作，只需重新启动Nginx服务。 sudo service nginx restart 让我们检查一下是否一切正常。 在web浏览器中打开我们新创建的网站。 请记住，我们使用了：81端口。

Minecraft
#

创建专用用户
#

sudo adduser --system --group minecraft
sudo mkdir -p /mnt/nas/minecraft
sudo chown minecraft:minecraft /mnt/nas/minecraft

下载 PaperMC
#

一款广泛使用、高性能的《我的世界》服务器,旨在修复游戏玩法和机制的不一致之处。

cd /mnt/nas/minecraft
sudo -u minecraft wget -O paper.jar https://api.papermc.io/v2/projects/paper/releases/latest/downloads/paper-1.20.4-latest.jar

首次启动
#

生成配置文件

sudo -u minecraft java -Xms1G -Xmx4G -jar paper.jar nogui
# 首次启动会失败（需同意 EULA），编辑 eula.txt
sudo -u minecraft sed -i 's/false/true/g' eula.txt

创建 systemd 服务
#

sudo nano /etc/systemd/system/minecraft.service

[Unit]
Description=Minecraft Server (PaperMC)
After=network.target

[Service]
Type=simple
User=minecraft
WorkingDirectory=/mnt/nas/minecraft
ExecStart=/usr/bin/java -Xms1G -Xmx3G -jar paper.jar nogui
Restart=on-failure
RestartSec=5

[Install]
WantedBy=multi-user.target

启用服务
#

sudo systemctl daemon-reload
sudo systemctl enable --now minecraft
sudo systemctl status minecraft

自动备份
#

每日一次 编辑文件

sudo nano /etc/cron.daily/minecraft-backup

#!/bin/bash
# 停止服务器（可选，或使用插件热备份）
sudo systemctl stop minecraft
# 备份 world 目录
tar -czf /mnt/nas/backups/minecraft_$(date +%F).tar.gz -C /mnt/nas/minecraft world
# 启动服务器
sudo systemctl start minecraft

赋予执行权限

sudo chmod +x /etc/cron.daily/minecraft-backup

第三方起动器
#

需要服务器关闭正版验证

  1. 编辑服务器配置文件
nano /mnt/nas/minecraft/server.properties
  1. 修改或添加
online-mode=false
  1. 重启服务器
sudo systemctl restart minecraft
